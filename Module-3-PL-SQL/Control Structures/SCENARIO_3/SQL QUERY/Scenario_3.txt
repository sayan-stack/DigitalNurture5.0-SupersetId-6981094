Scenario 3: Send Loan Due Reminders to Customers

Question: Write a PL/SQL block that fetches all loans due within the next 30 days and prints a reminder message for each customer.



DECLARE
    CURSOR loan_cur IS
    SELECT c.Name,
           l.LoanID,
           l.EndDate
    FROM Customers c
    JOIN Loans l
    ON c.CustomerID = l.CustomerID
    WHERE l.EndDate BETWEEN SYSDATE AND SYSDATE + 30;

BEGIN

    FOR rec IN loan_cur LOOP

        DBMS_OUTPUT.PUT_LINE(
            'Reminder: Dear ' || rec.Name ||
            ', your Loan ID ' || rec.LoanID ||
            ' is due on ' ||
            TO_CHAR(rec.EndDate,'DD-MON-YYYY')
        );

    END LOOP;

END;
